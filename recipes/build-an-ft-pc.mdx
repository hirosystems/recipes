```typescript
import { Pc } from '@stacks/transactions';

const postCondition = Pc.principal("<owner-principal-address>")
  // !hover will-send-gte
  .willSendGte(500) // using the tokens decimals
  .ft("<ft-contract-address>.token-ft", "<token-name>");
```
