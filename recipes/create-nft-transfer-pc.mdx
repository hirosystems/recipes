```typescript
import { Pc, Cl } from '@stacks/transactions';

const postCondition = Pc.principal("<owner-principal-address>")
  // !hover will-not-send-asset
  .willNotSendAsset()
  .nft('<nft-contract-address>.token-nft::token', Cl.uint(12));
```
