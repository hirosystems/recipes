```typescript
import { getPublicKeyFromPrivate } from "@stacks/encryption";
import { Cl, makeUnsignedSTXTokenTransfer } from "@stacks/transactions";

const privateKey = "<your-private-key>"; // KEEP THIS SECRET!

const publicKey = getPublicKeyFromPrivate(privateKey);

// !mark[/makeUnsignedSTXTokenTransfer/]
const transaction = await makeUnsignedSTXTokenTransfer({
  network: "testnet",
  recipient: Cl.standardPrincipal("<recipient-address>"),
  amount: 42_000_000n,
  fee: 500_000n,
  memo: "<message-here>", // optional
  publicKey,
});
```
